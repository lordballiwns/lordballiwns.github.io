function getTodayKey() { const date = new Date(); return `${date.getFullYear()}-${date.getMonth()}-${date.getDate()}`; } function getNextArm() { const lastArm = localStorage.getItem("lastArm") || "izquierdo"; return lastArm === "izquierdo" ? "derecho" : "izquierdo"; } function updateArm() { const todayKey = getTodayKey(); const storedDate = localStorage.getItem("date"); if (storedDate !== todayKey) { const newArm = getNextArm(); localStorage.setItem("lastArm", newArm); localStorage.setItem("date", todayKey); } return localStorage.getItem("lastArm"); } const arm = updateArm(); document.getElementById("arriba").textContent = arm === "izquierdo" ? `Brazo: izquierdo` : ""; document.getElementById("abajo").textContent = arm === "derecho" ? `Brazo: derecho` : "";
